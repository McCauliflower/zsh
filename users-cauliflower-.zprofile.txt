# ~/.zprofile

echo "Sourcing ~/.zprofile"

# Source the shared zshrc
if [ -f /Users/Shared/.zshrc ]; then
  source /Users/Shared/.zshrc
fi

# LOCAL OVERRIDES

# Add user-specific PATH entries:
export PATH="$HOME/.local/bin:/opt/homebrew/sbin:$PATH"

# set homebrew envs 
eval "$(/opt/homebrew/bin/brew shellenv)"

# Path to your Oh My Zsh installation
export ZSH="/Users/cauliflower/.oh-my-zsh"

# Plugins
plugins=(git git-auto-fetch history aliases alias-finder aws battery nvm nodenv zsh-interactive-cd zsh-navigation-tools)

# Source Oh My Zsh
source $ZSH/oh-my-zsh.sh

# Load fzf if available
[ -f ~/.fzf.zsh ] && source ~/.fzf.zsh
